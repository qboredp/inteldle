CREATE DATABASE inteldle CHARACTER SET utf8mb4;
USE inteldle;

CREATE TABLE usuarios(
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL,
    tipo ENUM('usuario','admin') DEFAULT 'usuario',
    criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE defeitos(
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255),
    descricao TEXT,
    imagem VARCHAR(255),
    tipo ENUM('hardware','software')
);

CREATE TABLE solucoes(
    id INT AUTO_INCREMENT PRIMARY KEY,
    defeito_id INT,
    componente VARCHAR(255),
    correto BOOLEAN DEFAULT FALSE,
    mensagem TEXT,
    FOREIGN KEY(defeito_id) REFERENCES defeitos(id)
);

CREATE TABLE ranking_diario(
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT,
    pontuacao INT DEFAULT 0,
    data_jogo DATE,
    FOREIGN KEY(usuario_id) REFERENCES usuarios(id)
);

CREATE TABLE ranking_infinito(
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT,
    pontuacao INT DEFAULT 0,
    FOREIGN KEY(usuario_id) REFERENCES usuarios(id)
);

CREATE TABLE historico(
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT,
    defeito_id INT,
    acertou BOOLEAN,
    pontos INT,
    data_partida TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
